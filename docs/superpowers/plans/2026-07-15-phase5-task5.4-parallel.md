 * @file tuple_queue.h
 * @brief 线程安全元组队列
 *
 * 用于并行 worker 间传输 TupleTableSlot 的多生产者/单消费者队列。

#ifndef DB_SQL_TUPLE_QUEUE_H
#define DB_SQL_TUPLE_QUEUE_H

#include "db/sql/nodes/nodetags.h"
#include <stdbool.h>
#include <pthread.h>

#ifdef __cplusplus
extern "C" {
#endif

/* ========================================================================
 * TupleQueueNode - 队列节点
 * ======================================================================== */

typedef struct TupleQueueNode {
    void                 *slot;              /* TupleTableSlot 指针 */
    struct TupleQueueNode *next;             /* 链表下一个节点 */
} TupleQueueNode;

/* ========================================================================
 * TupleQueue - 元组队列
 * ======================================================================== */

typedef struct TupleQueue {
    pthread_mutex_t      lock;               /* 互斥锁 */
    pthread_cond_t       not_empty;          /* 非空条件变量 */
    pthread_cond_t       not_full;           /* 非满条件变量 */
    TupleQueueNode       *head;              /* 队列头 */
    TupleQueueNode       *tail;              /* 队列尾 */
    int                  size;               /* 当前大小 */
    int                  capacity;           /* 最大容量 */
    bool                 closed;             /* 是否已关闭 */
} TupleQueue;

/* ========================================================================
 * 公共 API
 * ======================================================================== */

/**
 * @brief 创建元组队列
 *
 * @param capacity 最大容量（0 表示无限制）
 * @return 新创建的 TupleQueue；失败返回 NULL
 */
TupleQueue *TupleQueueCreate(int capacity);

/**
 * @brief 销毁元组队列
 *
 * @param queue 队列（可为 NULL）
 */
void TupleQueueDestroy(TupleQueue *queue);

/**
 * @brief 发送元组到队列
 *
 * 如果队列满则阻塞，直到有空间或队列关闭。
 *
 * @param queue 队列
 * @param slot  元组槽
 *
 * @return 0 成功；-1 队列已关闭
 */
int TupleQueueSend(TupleQueue *queue, void *slot);

/**
 * @brief 从队列接收元组
 *
 * 如果队列空则阻塞，直到有数据或队列关闭。
 *
 * @param queue 队列
 *
 * @return 元组槽；队列关闭且无数据返回 NULL
 */
void *TupleQueueReceive(TupleQueue *queue);

/**
 * @brief 关闭队列
 *
 * 通知所有等待的 producer/consumer 退出。
 *
 * @param queue 队列
 */
void TupleQueueClose(TupleQueue *queue);

#ifdef __cplusplus
}
#endif

/**
 * @file tuple_queue.c
 * @brief 线程安全元组队列实现
 */

#include "db/sql/tuple_queue.h"
#include <stdlib.h>

TupleQueue *TupleQueueCreate(int capacity)
{
    TupleQueue *queue = (TupleQueue *)calloc(1, sizeof(TupleQueue));
    if (!queue) {
        return NULL;
    }

    pthread_mutex_init(&queue->lock, NULL);
    pthread_cond_init(&queue->not_empty, NULL);
    pthread_cond_init(&queue->not_full, NULL);
    queue->capacity = capacity;
    queue->closed = false;

    return queue;
}

void TupleQueueDestroy(TupleQueue *queue)
{
    TupleQueueNode *node, *next;

    if (!queue) {
        return;
    }

    /* 释放所有节点 */
    node = queue->head;
    while (node) {
        next = node->next;
        free(node);
        node = next;
    }

    pthread_mutex_destroy(&queue->lock);
    pthread_cond_destroy(&queue->not_empty);
    pthread_cond_destroy(&queue->not_full);
    free(queue);
}

int TupleQueueSend(TupleQueue *queue, void *slot)
{
    TupleQueueNode *node;

    if (!queue) {
        return -1;
    }

    node = (TupleQueueNode *)malloc(sizeof(TupleQueueNode));
    if (!node) {
        return -1;
    }
    node->slot = slot;
    node->next = NULL;

    pthread_mutex_lock(&queue->lock);

    /* 如果队列满且非无限制，等待 */
    while (queue->capacity > 0 && queue->size >= queue->capacity && !queue->closed) {
        pthread_cond_wait(&queue->not_full, &queue->lock);
    }

    if (queue->closed) {
        free(node);
        pthread_mutex_unlock(&queue->lock);
        return -1;
    }

    /* 入队 */
    if (queue->tail) {
        queue->tail->next = node;
    } else {
        queue->head = node;
    }
    queue->tail = node;
    queue->size++;

    /* 通知 consumer */
    pthread_cond_signal(&queue->not_empty);
    pthread_mutex_unlock(&queue->lock);

    return 0;
}

void *TupleQueueReceive(TupleQueue *queue)
{
    void *slot;
    TupleQueueNode *node;

    if (!queue) {
        return NULL;
    }

    pthread_mutex_lock(&queue->lock);

    /* 如果队列空，等待 */
    while (queue->head == NULL && !queue->closed) {
        pthread_cond_wait(&queue->not_empty, &queue->lock);
    }

    node = queue->head;
    if (!node) {
        pthread_mutex_unlock(&queue->lock);
        return NULL;
    }

    /* 出队 */
    queue->head = node->next;
    if (!queue->head) {
        queue->tail = NULL;
    }
    queue->size--;

    slot = node->slot;
    free(node);

    /* 通知 producer */
    if (queue->capacity > 0) {
        pthread_cond_signal(&queue->not_full);
    }

    pthread_mutex_unlock(&queue->lock);
    return slot;
}

void TupleQueueClose(TupleQueue *queue)
{
    if (!queue) {
        return;
    }

    pthread_mutex_lock(&queue->lock);
    queue->closed = true;
    pthread_cond_broadcast(&queue->not_empty);
    pthread_cond_broadcast(&queue->not_full);
    pthread_mutex_unlock(&queue->lock);
}

/**
 * @file worker_pool.h
 * @brief Worker 线程池
 *
 * 管理多个并行 worker 线程的生命周期，用于并行执行计划子树。

#ifndef DB_SQL_WORKER_POOL_H
#define DB_SQL_WORKER_POOL_H

#include "db/sql/tuple_queue.h"
#include "db/sql/parallel.h"
#include <pthread.h>
#include <stdbool.h>

#ifdef __cplusplus
extern "C" {
#endif

/* ========================================================================
 * WorkerTask - 工作线程任务
 * ======================================================================== */

typedef struct WorkerTask {
    int               worker_id;         /* worker ID */
    void             *plan_state;        /* 要执行的 PlanState */
    TupleQueue       *output_queue;      /* 输出队列 */
    ParallelContext  *pcxt;             /* 并行上下文 */
} WorkerTask;

/* ========================================================================
 * WorkerPool - 线程池
 * ======================================================================== */

typedef struct WorkerPool {
    int               nworkers;          /* worker 数量 */
    pthread_t        *threads;           /* 线程数组 */
    WorkerTask       *tasks;             /* 任务数组 */
    volatile bool     shutdown;          /* 关闭标志 */
} WorkerPool;

/* ========================================================================
 * 公共 API
 * ======================================================================== */

/**
 * @brief 创建线程池
 *
 * @param nworkers worker 数量
 * @return 新创建的 WorkerPool；失败返回 NULL
 */
WorkerPool *WorkerPoolCreate(int nworkers);

/**
 * @brief 启动并行任务
 *
 * 每个 worker 执行对应的 task。
 *
 * @param pool   线程池
 * @param tasks  任务数组（长度 >= pool->nworkers）
 *
 * @return 0 成功；-1 失败
 */
int WorkerPoolStart(WorkerPool *pool, WorkerTask *tasks);

/**
 * @brief 等待所有 worker 完成
 *
 * @param pool 线程池
 */
void WorkerPoolWait(WorkerPool *pool);

/**
 * @brief 销毁线程池
 *
 * @param pool 线程池（可为 NULL）
 */
void WorkerPoolDestroy(WorkerPool *pool);

#ifdef __cplusplus
}
#endif

#endif /* DB_SQL_WORKER_POOL_H */

/**
 * @file worker_pool.c
 * @brief Worker 线程池实现
 */

#include "db/sql/worker_pool.h"
#include <stdlib.h>

/**
 * @brief Worker 线程入口函数
 */
static void *worker_main(void *arg)
{
    WorkerTask *task = (WorkerTask *)arg;
    TupleTableSlot *(*exec_proc)(void *) = NULL;
    TupleTableSlot *slot;

    /* 获取 PlanState 的执行函数 */
    if (task->plan_state) {
        void **ps = (void **)task->plan_state;
        exec_proc = (TupleTableSlot *(*)(void *))ps[3];  /* ExecProcNode 偏移 */
    }

    if (!exec_proc) {
        return NULL;
    }

    /* 循环执行，发送结果到输出队列 */
    while (!task->pcxt->coordinator->finished) {
        slot = exec_proc(task->plan_state);
        if (!slot) {
            break;
        }
        TupleQueueSend(task->output_queue, slot);
    }

    return NULL;
}

WorkerPool *WorkerPoolCreate(int nworkers)
{
    WorkerPool *pool;

    if (nworkers <= 0) {
        return NULL;
    }

    pool = (WorkerPool *)calloc(1, sizeof(WorkerPool));
    if (!pool) {
        return NULL;
    }

    pool->nworkers = nworkers;
    pool->threads = (pthread_t *)calloc(nworkers, sizeof(pthread_t));
    pool->tasks = (WorkerTask *)calloc(nworkers, sizeof(WorkerTask));

    if (!pool->threads || !pool->tasks) {
        free(pool->threads);
        free(pool->tasks);
        free(pool);
        return NULL;
    }

    return pool;
}

int WorkerPoolStart(WorkerPool *pool, WorkerTask *tasks)
{
    if (!pool || !tasks) {
        return -1;
    }

    pool->shutdown = false;

    for (int i = 0; i < pool->nworkers; i++) {
        pool->tasks[i] = tasks[i];
        if (pthread_create(&pool->threads[i], NULL, worker_main, &pool->tasks[i]) != 0) {
            return -1;
        }
    }

    return 0;
}

void WorkerPoolWait(WorkerPool *pool)
{
    if (!pool) {
        return;
    }

    for (int i = 0; i < pool->nworkers; i++) {
        pthread_join(pool->threads[i], NULL);
    }
}

void WorkerPoolDestroy(WorkerPool *pool)
{
    if (!pool) {
        return;
    }

    pool->shutdown = true;
    WorkerPoolWait(pool);

    free(pool->threads);
    free(pool->tasks);
    free(pool);
}

#include "db/sql/nodeGather.h"
#include "db/sql/worker_pool.h"
#include "db/sql/tuple_queue.h"
#include "db/sql/executor.h"
#include <stdlib.h>

/* ========================================================================
 * Gather 节点实现（真实并行）
 * ======================================================================== */

/**
 * @brief 懒初始化：启动并行 worker
 */
static void gather_init_real(GatherState *state)
{
    if (state->initialized) {
        return;
    }

    if (!state->pcxt || state->num_workers <= 0) {
        /* 没有 worker，退化为串行 */
        state->need_to_scan_locally = true;
        state->initialized = true;
        return;
    }

    /* 创建线程池 */
    state->worker_pool = WorkerPoolCreate(state->num_workers);
    if (!state->worker_pool) {
        state->need_to_scan_locally = true;
        state->initialized = true;
        return;
    }

    /* 创建输出队列 */
    state->output_queue = TupleQueueCreate(64);

    /* 准备 worker 任务 */
    WorkerTask *tasks = (WorkerTask *)calloc(state->num_workers, sizeof(WorkerTask));
    if (!tasks) {
        WorkerPoolDestroy(state->worker_pool);
        state->need_to_scan_locally = true;
        state->initialized = true;
        return;
    }

    for (int i = 0; i < state->num_workers; i++) {
        tasks[i].worker_id = i;
        tasks[i].plan_state = state->ps.lefttree;
        tasks[i].output_queue = state->output_queue;
        tasks[i].pcxt = state->pcxt;
    }

    /* 启动 worker */
    LaunchParallelWorkers(state->pcxt);
    WorkerPoolStart(state->worker_pool, tasks);

    state->reader = 0;
    state->initialized = true;
    free(tasks);
}

/**
 * @brief 从并行 worker 收集结果
 */
static TupleTableSlot *gather_getnext_parallel(GatherState *state)
{
    if (state->need_to_scan_locally) {
        /* 退化为串行 */
        if (state->ps.lefttree) {
            return ExecProcNode(state->ps.lefttree);
        }
        return NULL;
    }

    /* 从队列接收 worker 的结果 */
    TupleTableSlot *slot = (TupleTableSlot *)TupleQueueReceive(state->output_queue);
    return slot;
}

TupleTableSlot *ExecGather(GatherState *node)
{
    if (!node) {
        return NULL;
    }

    /* 懒初始化 */
    if (!node->initialized) {
        gather_init_real(node);
    }

    return gather_getnext_parallel(node);
}

void ExecEndGather(GatherState *node)
{
    if (!node) {
        return;
    }

    /* 结束子计划 */
    if (node->ps.lefttree) {
        ExecEndNode(node->ps.lefttree);
    }

    /* 等待 worker 完成 */
    if (node->worker_pool) {
        WorkerPoolWait(node->worker_pool);
    }

    /* 关闭输出队列 */
    if (node->output_queue) {
        TupleQueueClose(node->output_queue);
    }

    /* 释放并行上下文 */
    if (node->pcxt) {
        DestroyParallelContext(node->pcxt);
    }

    /* 释放线程池 */
    if (node->worker_pool) {
        WorkerPoolDestroy(node->worker_pool);
    }

    /* 释放结果槽 */
    if (node->ps.ps_ResultTupleSlot) {
        FreeTupleTableSlot(node->ps.ps_ResultTupleSlot);
    }

    /* 释放队列 */
    if (node->output_queue) {
        TupleQueueDestroy(node->output_queue);
    }
}

/* 替换 LaunchParallelWorkers 为真实启动 */
int LaunchParallelWorkers(ParallelContext *pcxt)
{
    if (!pcxt) {
        return 0;
    }

    /* 更新 worker 状态 */
    pcxt->nworkers_launched = pcxt->nworkers_to_launch;
    pcxt->coordinator->nworkers_attached = pcxt->nworkers_launched;

    return pcxt->nworkers_launched;
}

/* 替换 WaitForParallelWorkers 为真实等待 */
void WaitForParallelWorkers(ParallelContext *pcxt)
{
    if (!pcxt) {
        return;
    }

    /* 标记完成（真正的等待由 WorkerPoolWait 处理） */
    if (pcxt->coordinator) {
        pcxt->coordinator->finished = true;
    }
}
