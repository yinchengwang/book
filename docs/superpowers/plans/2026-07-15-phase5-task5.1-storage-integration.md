```c
/**
 * @file nodeSeqscan.c
 * @brief SeqScan 执行器节点实现
 *
 * 对接存储引擎，实现真正的全表扫描。
 */

#include "db/sql/nodeSeqscan.h"
#include "db/rel.h"
#include "db/heapam.h"
#include "db/storage.h"
#include <stdlib.h>

/* ========================================================================
 * SeqScan 执行器节点实现
 * ======================================================================== */

/**
 * @brief 执行 SeqScan 迭代
 */
TupleTableSlot *ExecSeqScan(PlanState *pstate)
{
    SeqScanState *node = (SeqScanState *)pstate;
    HeapTuple tuple;

    if (!node->ss_currentScanDesc) {
        return NULL;
    }

    /* 从存储层获取下一个元组 */
    tuple = heap_getnext(node->ss_currentScanDesc);

    if (tuple == NULL) {
        /* 扫描结束 */
        return NULL;
    }

    /* 将 HeapTuple 存储到 TupleTableSlot */
    ExecStoreHeapTuple(tuple, node->ss_ScanTupleSlot, false);

    return node->ss_ScanTupleSlot;
}

/**
 * @brief 初始化 SeqScan 执行状态
 */
PlanState *ExecInitSeqScan(Plan *plan, EState *estate, int eflags)
{
    SeqScanState *state;
    SeqScan *node = (SeqScan *)plan;

    (void)eflags;

    /* 分配状态结构 */
    state = (SeqScanState *)palloc0(estate->es_query_cxt, sizeof(SeqScanState));
    if (!state) {
        return NULL;
    }

    /* 初始化基类 */
    state->ps.type = T_SeqScanState;
    state->ps.plan = plan;
    state->ps.state = estate;
    state->ps.ExecProcNode = ExecSeqScan;
    state->ps.lefttree = NULL;
    state->ps.righttree = NULL;

    /* 打开目标表 Relation */
    state->ss_currentRelation = relation_open(node->scanrelid, REL_OPEN_READ);
    if (!state->ss_currentRelation) {
        pfree(state);
        return NULL;
    }

    /* 开始扫描 */
    state->ss_currentScanDesc = heap_beginscan(state->ss_currentRelation,
                                                 estate->es_snapshot);
    if (!state->ss_currentScanDesc) {
        relation_close(state->ss_currentRelation, REL_OPEN_READ);
        pfree(state);
        return NULL;
    }

    /* 创建结果槽 */
    state->ss_ScanTupleSlot = MakeTupleTableSlotWithMCxt(estate->es_query_cxt);

    return &state->ps;
}

/**
 * @brief 结束 SeqScan 执行
 */
void ExecEndSeqScan(SeqScanState *node)
{
    if (!node) {
        return;
    }

    /* 结束扫描 */
    if (node->ss_currentScanDesc) {
        heap_endscan(node->ss_currentScanDesc);
        node->ss_currentScanDesc = NULL;
    }

    /* 关闭 Relation */
    if (node->ss_currentRelation) {
        relation_close(node->ss_currentRelation, REL_OPEN_READ);
        node->ss_currentRelation = NULL;
    }

    /* 释放结果槽 */
    if (node->ss_ScanTupleSlot) {
        FreeTupleTableSlot(node->ss_ScanTupleSlot);
    }
}

/**
 * @brief 创建 SeqScan 计划节点
 */
SeqScan *make_seqscan(Oid relid)
{
    SeqScan *node = (SeqScan *)palloc(sizeof(SeqScan));
    if (!node) {
        return NULL;
    }

    node->plan.type = T_SeqScan;
    node->scanrelid = relid;

    return node;
}
```

```c
/* 添加存储层头文件 */
#include "db/rel.h"
#include "db/heapam.h"
#include "db/storage.h"

/* 修改 ExecModifyTable 实现 */
static TupleTableSlot *exec_modifytable_impl(PlanState *pstate)
{
    ModifyTableState *node = (ModifyTableState *)pstate;
    ModifyTableExtState *ext;
    Relation rel;
    TupleTableSlot *slot;

    if (!pstate) return NULL;
    ext = node->ext_state;
    if (!ext || ext->mt_finished) return NULL;

    /* 获取目标表 Relation */
    rel = relation_open(ext->relationOid, REL_OPEN_READWRITE);
    if (!rel) return NULL;

    /* 从子节点获取输入元组 */
    if (pstate->lefttree) {
        slot = ExecProcNode(pstate->lefttree);
        if (TupIsNull(slot)) {
            ext->mt_finished = true;
            relation_close(rel, REL_OPEN_READWRITE);
            return NULL;
        }
    } else {
        slot = node->mt_scan_slot;
        if (!slot) {
            ext->mt_finished = true;
            relation_close(rel, REL_OPEN_READWRITE);
            return NULL;
        }
    }

    /* 根据 DML 类型调用存储层 API */
    switch (ext->operation) {
        case CMD_INSERT:
            heap_insert(rel, slot->tts_tuple);
            ext->mt_inserted++;
            break;
        case CMD_UPDATE:
            heap_update(rel, &slot->tts_tid, slot->tts_tuple);
            ext->mt_updated++;
            break;
        case CMD_DELETE:
            heap_delete(rel, &slot->tts_tid);
            ext->mt_deleted++;
            break;
        default:
            break;
    }

    ext->mt_processed++;
    relation_close(rel, REL_OPEN_READWRITE);
    
    return slot;
}
```

```cpp
/**
 * @file test_storage_integration.cpp
 * @brief 存储引擎集成测试
 */

#include <gtest/gtest.h>
#include "db/sql/executor.h"
#include "db/sql/nodeSeqscan.h"
#include "db/sql/nodeModifyTable.h"
#include "db/rel.h"
#include "db/storage.h"

extern "C" {

class StorageIntegrationTest : public ::testing::Test {
protected:
    Relation users_table;
    EState *estate;
    MemoryContext ctx;

    void SetUp() override {
        ctx = AllocSetContextCreate(NULL, "IntegrationTest", 0, 8192, 8192);
        users_table = heap_create("users", ctx);
        estate = CreateExecutorState();
        estate->es_query_cxt = ctx;
    }

    void TearDown() override {
        FreeExecutorState(estate);
        heap_drop(users_table);
        MemoryContextDelete(ctx);
    }
};

/* 完整流程：INSERT → SELECT → DELETE */
TEST_F(StorageIntegrationTest, InsertSelectDelete) {
    /* Step 1: INSERT 3 行 */
    for (int i = 1; i <= 3; i++) {
        ModifyTablePlan *plan = make_modifytable_plan(CMD_INSERT, users_table->rd_id);
        PlanState *ps = ExecInitModifyTable((Plan *)plan, estate, 0);
        
        TupleTableSlot *slot = MakeTupleTableSlot();
        slot->tts_values[0] = Int32GetDatum(i);
        slot->tts_values[1] = CStringGetDatum("User");
        
        ((ModifyTableState *)ps)->mt_scan_slot = slot;
        ExecProcNode(ps);
        ExecEndNode(ps);
    }
    
    /* Step 2: SELECT 验证 3 行 */
    SeqScan *scan = make_seqscan(users_table->rd_id);
    PlanState *scan_ps = ExecInitSeqScan((Plan *)scan, estate, 0);
    
    int count = 0;
    while (ExecProcNode(scan_ps) != NULL) {
        count++;
    }
    EXPECT_EQ(count, 3);
    ExecEndNode(scan_ps);
    
    /* Step 3: DELETE 所有行 */
    ModifyTablePlan *del_plan = make_modifytable_plan(CMD_DELETE, users_table->rd_id);
    PlanState *del_ps = ExecInitModifyTable((Plan *)del_plan, estate, 0);
    ExecProcNode(del_ps);
    ExecEndNode(del_ps);
    
    /* Step 4: 验证删除后 0 行 */
    scan_ps = ExecInitSeqScan((Plan *)scan, estate, 0);
    count = 0;
    while (ExecProcNode(scan_ps) != NULL) {
        count++;
    }
    EXPECT_EQ(count, 0);
    ExecEndNode(scan_ps);
}

} /* extern "C" */
```
