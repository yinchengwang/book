```cpp
/**
 * @file benchmark_config.h
 * @brief 性能基准测试配置
 */

#ifndef BENCHMARK_CONFIG_H
#define BENCHMARK_CONFIG_H

#include <chrono>
#include <string>
#include <vector>
#include <fstream>
#include <iostream>

/* ========================================================================
 * BenchmarkConfig - 基准配置
 * ======================================================================== */

struct BenchmarkConfig {
    int      nrows;           /* 行数 */
    int      ncols;           /* 列数 */
    int      nworkers;        /* 并行 worker 数 */
    int      warmup_runs;      /* 预热次数 */
    int      measure_runs;     /* 测量次数 */
    bool     verbose;         /* 详细输出 */
    std::string output_file;  /* 结果输出文件 */
    
    BenchmarkConfig()
        : nrows(100000),
          ncols(10),
          nworkers(1),
          warmup_runs(3),
          measure_runs(5),
          verbose(false),
          output_file("benchmark_results.json") {}
};

/* ========================================================================
 * BenchmarkResult - 基准结果
 * ======================================================================== */

struct BenchmarkResult {
    std::string name;              /* 基准名称 */
    int64_t     duration_ms;       /* 平均耗时（毫秒） */
    int64_t     min_ms;            /* 最小耗时 */
    int64_t     max_ms;            /* 最大耗时 */
    int64_t     rows_processed;    /* 处理行数 */
    double      rows_per_sec;      /* 每秒行数 */
    bool        passed;            /* 是否达标 */
};

/* ========================================================================
 * BenchmarkRecorder - 结果记录器
 * ======================================================================== */

class BenchmarkRecorder {
public:
    void AddResult(const BenchmarkResult& result) {
        results_.push_back(result);
    }
    
    void PrintSummary() const {
        std::cout << "\n=== Benchmark Results ===\n";
        std::cout << "Name\t\tAvg(ms)\tMin(ms)\tMax(ms)\tRows/sec\tPassed\n";
        std::cout << "----\t\t-------\t-------\t-------\t--------\t------\n";
        
        for (const auto& r : results_) {
            std::cout << r.name << "\t\t"
                      << r.duration_ms << "\t"
                      << r.min_ms << "\t"
                      << r.max_ms << "\t"
                      << (int)r.rows_per_sec << "\t"
                      << (r.passed ? "YES" : "NO") << "\n";
        }
    }
    
    void WriteJson(const std::string& filename) const {
        std::ofstream out(filename);
        out << "{\n  \"benchmarks\": [\n";
        
        for (size_t i = 0; i < results_.size(); i++) {
            const auto& r = results_[i];
            out << "    {\n";
            out << "      \"name\": \"" << r.name << "\",\n";
            out << "      \"duration_ms\": " << r.duration_ms << ",\n";
            out << "      \"min_ms\": " << r.min_ms << ",\n";
            out << "      \"max_ms\": " << r.max_ms << ",\n";
            out << "      \"rows_processed\": " << r.rows_processed << ",\n";
            out << "      \"rows_per_sec\": " << r.rows_per_sec << ",\n";
            out << "      \"passed\": " << (r.passed ? "true" : "false") << "\n";
            out << "    }";
            if (i < results_.size() - 1) out << ",";
            out << "\n";
        }
        
        out << "  ]\n}\n";
        out.close();
    }
    
private:
    std::vector<BenchmarkResult> results_;
};

/* ========================================================================
 * BenchmarkTimer - 计时器
 * ======================================================================== */

class BenchmarkTimer {
public:
    void Start() {
        start_ = std::chrono::high_resolution_clock::now();
    }
    
    int64_t Stop() {
        auto end = std::chrono::high_resolution_clock::now();
        auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(end - start_);
        return duration.count();
    }
    
private:
    std::chrono::high_resolution_clock::time_point start_;
};

/* ========================================================================
 * BenchmarkRunner - 基准运行器
 * ======================================================================== */

template<typename Func>
BenchmarkResult RunBenchmark(const std::string& name,
                              Func&& func,
                              const BenchmarkConfig& cfg,
                              int64_t expected_max_ms = 0) {
    BenchmarkResult result;
    result.name = name;
    result.passed = true;
    
    std::vector<int64_t> durations;
    int64_t total_rows = 0;
    
    /* 预热 */
    for (int i = 0; i < cfg.warmup_runs; i++) {
        total_rows = func();
    }
    
    /* 测量 */
    BenchmarkTimer timer;
    for (int i = 0; i < cfg.measure_runs; i++) {
        timer.Start();
        total_rows = func();
        int64_t ms = timer.Stop();
        durations.push_back(ms);
    }
    
    /* 统计 */
    result.rows_processed = total_rows;
    result.min_ms = *std::min_element(durations.begin(), durations.end());
    result.max_ms = *std::max_element(durations.begin(), durations.end());
    
    int64_t sum = 0;
    for (auto d : durations) sum += d;
    result.duration_ms = sum / durations.size();
    
    result.rows_per_sec = (double)total_rows / (result.duration_ms / 1000.0);
    
    /* 达标检查 */
    if (expected_max_ms > 0 && result.duration_ms > expected_max_ms) {
        result.passed = false;
    }
    
    return result;
}

#endif /* BENCHMARK_CONFIG_H */
```

```cpp
/**
 * @file benchmark_main.cpp
 * @brief 基准测试入口
 */

#include "benchmark_config.h"
#include <gtest/gtest.h>

/* 全局配置 */
BenchmarkConfig g_config;
BenchmarkRecorder g_recorder;

/* 配置解析 */
void ParseArgs(int argc, char** argv) {
    for (int i = 1; i < argc; i++) {
        std::string arg = argv[i];
        if (arg == "--rows" && i + 1 < argc) {
            g_config.nrows = std::stoi(argv[++i]);
        } else if (arg == "--workers" && i + 1 < argc) {
            g_config.nworkers = std::stoi(argv[++i]);
        } else if (arg == "--output" && i + 1 < argc) {
            g_config.output_file = argv[++i];
        } else if (arg == "--verbose") {
            g_config.verbose = true;
        }
    }
}

/* 主函数 */
int main(int argc, char** argv) {
    ParseArgs(argc, argv);
    
    ::testing::InitGoogleTest(&argc, argv);
    int ret = RUN_ALL_TESTS();
    
    /* 打印摘要 */
    g_recorder.PrintSummary();
    
    /* 写入 JSON */
    g_recorder.WriteJson(g_config.output_file);
    
    return ret;
}
```

```cpp
        ctx = AllocSetContextCreate(NULL, "BenchmarkCtx", 0, 8192, 8192);

        heap_drop(test_rel);

        int64_t count = 0;
        TupleTableSlot *slot;
```

```cpp
/**
 * @file benchmark_vector.cpp
 * @brief 向量查询性能基准
 */

#include "benchmark_config.h"
#include "db/index/hnsw.h"
#include <gtest/gtest.h>
#include <random>

extern BenchmarkConfig g_config;
extern BenchmarkRecorder g_recorder;

class BenchmarkVectorTest : public ::testing::Test {
protected:
    HNSWIndex *index;
    float **vectors;
    int dim = 128;
    
    void SetUp() override {
        /* 创建 HNSW 索引 */
        index = hnsw_create(dim, 16, 200);
        
        /* 生成随机向量 */
        std::mt19937 rng(42);
        std::normal_distribution<float> dist(0.0f, 1.0f);
        
        vectors = new float*[g_config.nrows];
        for (int i = 0; i < g_config.nrows; i++) {
            vectors[i] = new float[dim];
            for (int d = 0; d < dim; d++) {
                vectors[i][d] = dist(rng);
            }
            hnsw_insert(index, i, vectors[i]);
        }
    }
    
    void TearDown() override {
        hnsw_destroy(index);
        for (int i = 0; i < g_config.nrows; i++) {
            delete[] vectors[i];
        }
        delete[] vectors;
    }
};

/* 向量搜索 K=10 */
TEST_F(BenchmarkVectorTest, SearchK10) {
    float query[128];
    for (int d = 0; d < dim; d++) query[d] = 0.5f;
    
    auto func = [&]() -> int64_t {
        int64_t total_queries = 100;
        for (int q = 0; q < total_queries; q++) {
            int *results = hnsw_search(index, query, 10);
            free(results);
        }
        return total_queries;
    };
    
    BenchmarkResult result = RunBenchmark("VectorSearchK10", func, g_config);
    result.rows_per_sec = result.rows_processed * 1000.0 / result.duration_ms;
    g_recorder.AddResult(result);
    
    EXPECT_GT(result.rows_per_sec, 1000) << "Vector search QPS should > 1000";
}

/* 向量搜索 K=100 */
TEST_F(BenchmarkVectorTest, SearchK100) {
    float query[128];
    for (int d = 0; d < dim; d++) query[d] = 0.5f;
    
    auto func = [&]() -> int64_t {
        int64_t total_queries = 100;
        for (int q = 0; q < total_queries; q++) {
            int *results = hnsw_search(index, query, 100);
            free(results);
        }
        return total_queries;
    };
    
    BenchmarkResult result = RunBenchmark("VectorSearchK100", func, g_config);
    result.rows_per_sec = result.rows_processed * 1000.0 / result.duration_ms;

}
```
