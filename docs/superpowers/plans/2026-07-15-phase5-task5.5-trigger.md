# Phase 5 Task 5.5：触发器调用机制

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-step. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 实现触发器函数的注册和调用机制，使 BEFORE/AFTER 触发器能真正执行业务逻辑。

**Architecture:** 
- TriggerFunctionRegistry：全局触发器函数注册表
- TriggerFunc 函数指针类型：统一触发器函数签名
- ExecTriggerFunction：查找并调用已注册的触发器函数

**Tech Stack:** C11, MemoryContext, 内置触发器函数示例

- 所有测试必须通过
- 触发器函数必须符合 TriggerFunc 签名
- BEFORE 触发器可修改元组，AFTER 触发器仅通知

### Step 1: Write the header

**File:** `engineering/include/db/sql/trigger_functions.h`

```c
/**
 * @file trigger_functions.h
 * @brief 触发器函数注册与调用
 *
 * 提供触发器函数的注册表和调用机制。
 * 触发器函数统一签名为：TupleTableSlot *(*)(TriggerData *)
 */

#ifndef DB_SQL_TRIGGER_FUNCTIONS_H
#define DB_SQL_TRIGGER_FUNCTIONS_H

#include "db/sql/trigger.h"
#include <stdbool.h>

#ifdef __cplusplus
extern "C" {
#endif

/* ========================================================================
 * TriggerFunc - 触发器函数类型
 * ======================================================================== */

/**
 * @brief 触发器函数签名
 *
 * @param trig_data 触发器数据
 *
 * @return 处理后的元组槽；NULL 表示跳过操作
 */
typedef TupleTableSlot *(*TriggerFunc)(TriggerData *trig_data);

/* ========================================================================
 * TriggerFunctionEntry - 注册表条目
 * ======================================================================== */

typedef struct TriggerFunctionEntry {
    char           *func_name;          /* 函数名 */
    Oid             func_oid;           /* 函数 OID */
    TriggerFunc     func_ptr;           /* 函数指针 */
    bool            is_builtin;         /* 是否内置 */
} TriggerFunctionEntry;

/* ========================================================================
 * 公共 API
 * ======================================================================== */

/**
 * @brief 初始化触发器函数注册表
 */
void InitTriggerFunctionRegistry(void);

/**
 * @brief 注册触发器函数
 *
 * @param name   函数名
 * @param oid    函数 OID
 * @param func   函数指针
 * @param builtin 是否内置
 *
 * @return 0 成功；-1 失败
 */
int RegisterTriggerFunction(const char *name, Oid oid, TriggerFunc func, bool builtin);

/**
 * @brief 查找触发器函数
 *
 * @param oid 函数 OID
 *
 * @return 函数指针；未找到返回 NULL
 */
TriggerFunc FindTriggerFunction(Oid oid);

/**
 * @brief 按名称查找触发器函数
 *
 * @param name 函数名
 *
 * @return 函数指针；未找到返回 NULL
 */
TriggerFunc FindTriggerFunctionByName(const char *name);

/**
 * @brief 注册内置触发器函数
 *
 * 注册系统预定义的触发器函数。
 */
void RegisterBuiltinTriggerFunctions(void);

#ifdef __cplusplus
}
#endif

#endif /* DB_SQL_TRIGGER_FUNCTIONS_H */
```

### Step 2: Write the implementation

**File:** `engineering/src/db/sql/trigger_functions.c`

```c
/**
 * @file trigger_functions.c
 * @brief 触发器函数注册与调用实现
 */

#include "db/sql/trigger_functions.h"
#include "db/sql/memctx.h"
#include <stdlib.h>
#include <string.h>

/* ========================================================================
 * 全局注册表
 * ======================================================================== */

#define MAX_TRIGGER_FUNCTIONS 128

static TriggerFunctionEntry g_trigger_registry[MAX_TRIGGER_FUNCTIONS];
static int g_registry_count = 0;
static bool g_registry_initialized = false;

/* ========================================================================
 * 内置触发器函数
 * ======================================================================== */

/**
 * @brief 自动时间戳触发器
 *
 * BEFORE INSERT 时自动设置 created_at 列。
 */
TupleTableSlot *trigger_auto_timestamp(TriggerData *trig_data)
{
    TupleTableSlot *slot;
    Datum now;

    if (!trig_data || !trig_data->tg_trigslot) {
        return NULL;
    }

    slot = trig_data->tg_trigslot;
    now = Int64GetDatum(time(NULL));

    /* 查找 created_at 列并设置 */
    for (int i = 0; i < slot->tts_nvalid; i++) {
        if (slot->tts_tupleDesc &&
            strcmp(slot->tts_tupleDesc->attrs[i].attname, "created_at") == 0) {
            slot->tts_values[i] = now;
            slot->tts_isnull[i] = false;
            break;
        }
    }

    return slot;
}

/**
 * @brief 审计日志触发器
 *
 * AFTER INSERT/UPDATE/DELETE 时记录操作日志。
 */
TupleTableSlot *trigger_audit_log(TriggerData *trig_data)
{
    const char *operation;

    if (!trig_data) {
        return NULL;
    }

    /* 确定操作类型 */
    if (TRIGGER_FIRED_BEFORE_INSERT(trig_data->tg_event)) {
        operation = "INSERT";
    } else if (TRIGGER_FIRED_BEFORE_UPDATE(trig_data->tg_event)) {
        operation = "UPDATE";
    } else if (TRIGGER_FIRED_BEFORE_DELETE(trig_data->tg_event)) {
        operation = "DELETE";
    } else {
        operation = "OTHER";
    }

    /* 写入审计日志（简化：打印到 stdout） */
    printf("[AUDIT] Operation: %s, Table: %u\n",
           operation, trig_data->tg_desc->tgrelid);

    return trig_data->tg_trigslot;
}

/**
 * @brief 数据校验触发器
 *
 * BEFORE INSERT/UPDATE 时验证数据完整性。
 */
TupleTableSlot *trigger_validate_data(TriggerData *trig_data)
{
    TupleTableSlot *slot;

    if (!trig_data || !trig_data->tg_trigslot) {
        return NULL;
    }

    slot = trig_data->tg_trigslot;

    /* 验证所有非空列 */
    for (int i = 0; i < slot->tts_nvalid; i++) {
        if (!slot->tts_tupleDesc->attrs[i].attnotnull &&
            slot->tts_isnull[i]) {
            /* 列不允许 NULL */
            return NULL;
        }
    }

    return slot;
}

/* ========================================================================
 * 公共 API 实现
 * ======================================================================== */

void InitTriggerFunctionRegistry(void)
{
    if (g_registry_initialized) {
        return;
    }

    memset(g_trigger_registry, 0, sizeof(g_trigger_registry));
    g_registry_count = 0;
    g_registry_initialized = true;

    /* 注册内置函数 */
    RegisterBuiltinTriggerFunctions();
}

int RegisterTriggerFunction(const char *name, Oid oid, TriggerFunc func, bool builtin)
{
    if (g_registry_count >= MAX_TRIGGER_FUNCTIONS) {
        return -1;
    }

    TriggerFunctionEntry *entry = &g_trigger_registry[g_registry_count];
    entry->func_name = strdup(name);
    entry->func_oid = oid;
    entry->func_ptr = func;
    entry->is_builtin = builtin;

    g_registry_count++;
    return 0;
}

TriggerFunc FindTriggerFunction(Oid oid)
{
    for (int i = 0; i < g_registry_count; i++) {
        if (g_trigger_registry[i].func_oid == oid) {
            return g_trigger_registry[i].func_ptr;
        }
    }
    return NULL;
}

TriggerFunc FindTriggerFunctionByName(const char *name)
{
    for (int i = 0; i < g_registry_count; i++) {
        if (strcmp(g_trigger_registry[i].func_name, name) == 0) {
            return g_trigger_registry[i].func_ptr;
        }
    }
    return NULL;
}

void RegisterBuiltinTriggerFunctions(void)
{
    /* 注册内置触发器函数 */
    RegisterTriggerFunction("auto_timestamp", 1001, trigger_auto_timestamp, true);
    RegisterTriggerFunction("audit_log", 1002, trigger_audit_log, true);
    RegisterTriggerFunction("validate_data", 1003, trigger_validate_data, true);
}
```

### Step 1: Modify ExecTrigger to call real function

**File:** `engineering/src/db/sql/trigger.c`

```c
/* 添加头文件 */
#include "db/sql/trigger_functions.h"

/**
 * @brief 执行单个触发器
 *
 * 查找并调用已注册的触发器函数。
 */
TupleTableSlot *ExecTrigger(TriggerData *trig_data)
{
    TriggerDesc *desc;
    TriggerFunc func;
    TupleTableSlot *result;

    if (!trig_data || !trig_data->tg_desc) {
        return NULL;
    }

    desc = trig_data->tg_desc;

    /* 查找触发器函数 */
    func = FindTriggerFunction(desc->tgfuncid);
    if (!func) {
        /* 函数未注册，返回原槽（无操作） */
        return trig_data->tg_trigslot;
    }

    /* 调用触发器函数 */
    result = func(trig_data);

    return result;
}

/**
 * @brief 执行 BEFORE 触发器链
 */
TupleTableSlot *ExecBeforeTriggers(EState *estate,
                                   ResultRelInfo *rrinfo,
                                   TriggerEvent trig_event,
                                   TupleTableSlot *slot,
                                   TupleTableSlot *newslot)
{
    TriggerData data;
    TriggerDesc *desc;
    TupleTableSlot *result;

    (void)estate;
    (void)rrinfo;

    if (!slot) {
        return NULL;
    }

    /* 构造 TriggerData */
    memset(&data, 0, sizeof(data));
    data.type = T_TriggerData;
    data.tg_event = trig_event;
    data.tg_trigslot = slot;
    data.tg_newslot = newslot;

    /* 遍历 BEFORE 触发器链 */
    for (int i = 0; i < rrinfo->ri_TrigDesc->n_triggers; i++) {
        desc = rrinfo->ri_TrigDesc->triggers[i];
        if (!desc || !desc->tgenabled) {
            continue;
        }

        /* 检查是否匹配当前事件 */
        if (!TriggerMatchesEvent(desc, trig_event)) {
            continue;
        }

        if (!TriggerIsBefore(desc)) {
            continue;
        }

        data.tg_desc = desc;
        result = ExecTrigger(&data);

        /* 如果触发器返回 NULL，中止操作 */
        if (!result) {
            return NULL;
        }

        /* 更新 slot 供下一个触发器使用 */
        slot = result;
        data.tg_trigslot = slot;
    }

    return slot;
}

/**
 * @brief 执行 AFTER 触发器链
 */
void ExecAfterTriggers(EState *estate,
                       ResultRelInfo *rrinfo,
                       TriggerEvent trig_event,
                       TupleTableSlot *slot)
{
    TriggerData data;
    TriggerDesc *desc;

    (void)estate;

    if (!slot || !rrinfo->ri_TrigDesc) {
        return;
    }

    memset(&data, 0, sizeof(data));
    data.type = T_TriggerData;
    data.tg_event = trig_event;
    data.tg_trigslot = slot;

    for (int i = 0; i < rrinfo->ri_TrigDesc->n_triggers; i++) {
        desc = rrinfo->ri_TrigDesc->triggers[i];
        if (!desc || !desc->tgenabled) {
            continue;
        }

        if (!TriggerMatchesEvent(desc, trig_event)) {
            continue;
        }

        if (!TriggerIsAfter(desc)) {
            continue;
        }

        data.tg_desc = desc;
        ExecTrigger(&data);  /* 忽略 AFTER 触发器的返回值 */
    }
}
```
