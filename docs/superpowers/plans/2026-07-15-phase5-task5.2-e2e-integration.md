```c
/**
 * @file sql_driver.h
 * @brief SQL 执行驱动
 *
 * 提供从 SQL 文本到执行结果的统一入口。
 */

#ifndef DB_SQL_SQL_DRIVER_H
#define DB_SQL_SQL_DRIVER_H

#include "db/sql/executor.h"
#include "db/sql/expr.h"
#include <stdint.h>

#ifdef __cplusplus
extern "C" {
#endif

/* ========================================================================
 * QueryResult - 查询结果
 * ======================================================================== */

typedef struct QueryResult {
    int      nrows;           /* 行数 */
    int      ncols;           /* 列数 */
    char   **col_names;       /* 列名数组 */
    char  ***rows;            /* 行数据数组 */
    char  **error_msg;        /* 错误信息 */
} QueryResult;

/* ========================================================================
 * 公共 API
 * ======================================================================== */

/**
 * @brief 执行 SQL 语句
 *
 * @param sql SQL 文本
 * @param db  数据库上下文
 *
 * @return 查询结果（调用者需通过 FreeQueryResult 释放）
 */
QueryResult *execute_sql(const char *sql, void *db);

/**
 * @brief 释放查询结果
 *
 * @param result 查询结果
 */
void FreeQueryResult(QueryResult *result);

/**
 * @brief 执行 DDL 语句
 *
 * @param sql DDL 文本
 * @param db  数据库上下文
 *
 * @return 0 成功；-1 失败
 */
int execute_ddl(const char *sql, void *db);

#ifdef __cplusplus
}
#endif

#endif /* DB_SQL_SQL_DRIVER_H */
```

### Step 2: Write the implementation

**File:** `engineering/src/db/sql/sql_driver.c`

```c
/**
 * @file sql_driver.c
 * @brief SQL 执行驱动实现
 */

#include "db/sql/sql_driver.h"
#include "db/sql/planner.h"
#include "db/sql/executor.h"
#include "db/sql/expr.h"
#include "db/parser/sql/parse_analyze.h"
#include "db/catalog.h"
#include <stdlib.h>
#include <string.h>

/**
 * @brief 执行 SQL 语句
 */
QueryResult *execute_sql(const char *sql, void *db)
{
    QueryResult *result;
    Node *parsed_tree;
    Plan *plan;
    QueryDesc *query_desc;
    TupleTableSlot *slot;
    int nrows = 0, capacity = 64;
    char ***rows = NULL;

    (void)db;

    result = (QueryResult *)calloc(1, sizeof(QueryResult));
    if (!result) {
        return NULL;
    }

    /* Step 1: 解析 SQL */
    parsed_tree = parse_analyze(sql);
    if (!parsed_tree) {
        asprintf(&result->error_msg, "Parse error: %s", sql);
        return result;
    }

    /* Step 2: 计划生成 */
    plan = planner(parsed_tree);
    if (!plan) {
        asprintf(&result->error_msg, "Plan error: %s", sql);
        free(parsed_tree);
        return result;
    }

    /* Step 3: 执行计划 */
    query_desc = CreateQueryDesc(plan, NULL, NULL, NULL);
    if (!query_desc) {
        asprintf(&result->error_msg, "Executor init error");
        free(plan);
        free(parsed_tree);
        return result;
    }

    ExecutorStart(query_desc, 0);
    ExecutorRun(query_desc, forward);

    /* Step 4: 收集结果集 */
    rows = (char ***)calloc(capacity, sizeof(char **));
    while ((slot = ExecutorRunGetNext(query_desc)) != NULL) {
        if (nrows >= capacity) {
            capacity *= 2;
            rows = (char ***)realloc(rows, capacity * sizeof(char **));
        }
        rows[nrows] = (char **)calloc(slot->tts_nvalid, sizeof(char *));
        for (int i = 0; i < slot->tts_nvalid; i++) {
            if (slot->tts_isnull[i]) {
                rows[nrows][i] = strdup("NULL");
            } else {
                char buf[64];
                snprintf(buf, sizeof(buf), "%lld",
                         (long long)DatumGetInt64(slot->tts_values[i]));
                rows[nrows][i] = strdup(buf);
            }
        }
        nrows++;
    }

    ExecutorFinish(query_desc);
    ExecutorEnd(query_desc);

    /* Step 5: 填充结果 */
    result->nrows = nrows;
    result->ncols = 0;
    result->rows = rows;
    result->col_names = NULL;

    /* 获取列数 */
    if (nrows > 0 && rows[0] != NULL) {
        result->ncols = 0;
        while (rows[0][result->ncols] != NULL) {
            result->ncols++;
        }
    }

    free(plan);
    free(parsed_tree);

    return result;
}

/**
 * @brief 释放查询结果
 */
void FreeQueryResult(QueryResult *result)
{
    if (!result) return;

    if (result->rows) {
        for (int i = 0; i < result->nrows; i++) {
            if (result->rows[i]) {
                for (int j = 0; j < result->ncols; j++) {
                    free(result->rows[i][j]);
                }
                free(result->rows[i]);
            }
        }
        free(result->rows);
    }

    if (result->col_names) {
        for (int i = 0; i < result->ncols; i++) {
            free(result->col_names[i]);
        }
        free(result->col_names);
    }

    free(result->error_msg);
    free(result);
}

/**
 * @brief 执行 DDL 语句
 */
int execute_ddl(const char *sql, void *db)
{
    Node *parsed_tree;
    (void)db;

    parsed_tree = parse_analyze(sql);
    if (!parsed_tree) {
        return -1;
    }

    /* 简化实现：DDL 直接由解析器处理 */
    free(parsed_tree);
    return 0;
}
```

```cpp
/**
 * @file test_e2e_query.cpp
 * @brief SQL 端到端查询测试
 */

#include <gtest/gtest.h>
#include "db/sql/sql_driver.h"
#include "db/storage.h"

extern "C" {

class E2EQueryTest : public ::testing::Test {
protected:
    void *db;

    void SetUp() override {
        db = kv_open(":memory:");  /* 内存数据库 */
    }

    void TearDown() override {
        if (db) kv_close(db);
    }
};

/* 最简单的 SELECT */
TEST_F(E2EQueryTest, SimpleSelect) {
    QueryResult *result = execute_sql("SELECT 1", db);
    ASSERT_NE(result, nullptr);
    ASSERT_GT(result->nrows, 0);
    FreeQueryResult(result);
}

/* SELECT 常量表达式 */
TEST_F(E2EQueryTest, SelectConstant) {
    QueryResult *result = execute_sql("SELECT 1 + 2", db);
    ASSERT_NE(result, nullptr);
    ASSERT_GT(result->nrows, 0);
    FreeQueryResult(result);
}

/* INSERT 后 SELECT */
TEST_F(E2EQueryTest, InsertAndSelect) {
    execute_ddl("CREATE TABLE t (id INT, name TEXT)", db);
    execute_sql("INSERT INTO t VALUES (1, 'Alice')", db);
    execute_sql("INSERT INTO t VALUES (2, 'Bob')", db);
    
    QueryResult *result = execute_sql("SELECT * FROM t", db);
    ASSERT_NE(result, nullptr);
    EXPECT_EQ(result->nrows, 2);
    FreeQueryResult(result);
}

/* SELECT 带 WHERE 过滤 */
TEST_F(E2EQueryTest, SelectWithWhere) {
    execute_ddl("CREATE TABLE t (id INT, age INT)", db);
    execute_sql("INSERT INTO t VALUES (1, 20)", db);
    execute_sql("INSERT INTO t VALUES (2, 30)", db);
    execute_sql("INSERT INTO t VALUES (3, 15)", db);
    
    QueryResult *result = execute_sql("SELECT * FROM t WHERE age > 18", db);
    ASSERT_NE(result, nullptr);
    EXPECT_EQ(result->nrows, 2);
    FreeQueryResult(result);
}

/* SELECT 带聚合 */
TEST_F(E2EQueryTest, SelectWithAggregate) {
    execute_ddl("CREATE TABLE t (id INT, val INT)", db);
    execute_sql("INSERT INTO t VALUES (1, 10)", db);
    execute_sql("INSERT INTO t VALUES (2, 20)", db);
    execute_sql("INSERT INTO t VALUES (3, 30)", db);
    
    QueryResult *result = execute_sql("SELECT COUNT(*), SUM(val) FROM t", db);
    ASSERT_NE(result, nullptr);
    EXPECT_EQ(result->nrows, 1);
    FreeQueryResult(result);
}

/* 多表连接 */
TEST_F(E2EQueryTest, JoinQuery) {
    execute_ddl("CREATE TABLE users (id INT, name TEXT)", db);
    execute_ddl("CREATE TABLE orders (id INT, user_id INT, total INT)", db);
    
    execute_sql("INSERT INTO users VALUES (1, 'Alice')", db);
    execute_sql("INSERT INTO users VALUES (2, 'Bob')", db);
    execute_sql("INSERT INTO orders VALUES (1, 1, 100)", db);
    execute_sql("INSERT INTO orders VALUES (2, 1, 200)", db);
    
    QueryResult *result = execute_sql(
        "SELECT u.name, o.total FROM users u JOIN orders o ON u.id = o.user_id",
        db);
    ASSERT_NE(result, nullptr);
    EXPECT_GT(result->nrows, 0);
    FreeQueryResult(result);
}

} /* extern "C" */
```

```cpp
/**
 * @file test_e2e_dml.cpp
 * @brief SQL 端到端 DML 测试
 */

#include <gtest/gtest.h>
#include "db/sql/sql_driver.h"
#include "db/storage.h"

extern "C" {

class E2EDMLTest : public ::testing::Test {
protected:
    void *db;

    void SetUp() override {
        db = kv_open(":memory:");
        execute_ddl("CREATE TABLE t (id INT, val TEXT)", db);
    }

    void TearDown() override {
        if (db) kv_close(db);
    }
};

/* INSERT 单行 */
TEST_F(E2EDMLTest, InsertSingleRow) {
    int rc = execute_sql("INSERT INTO t VALUES (1, 'hello')", db);
    EXPECT_EQ(rc, 0);
}

/* INSERT 多行 */
TEST_F(E2EDMLTest, InsertMultipleRows) {
    execute_sql("INSERT INTO t VALUES (1, 'a')", db);
    execute_sql("INSERT INTO t VALUES (2, 'b')", db);
    execute_sql("INSERT INTO t VALUES (3, 'c')", db);
    
    QueryResult *result = execute_sql("SELECT COUNT(*) FROM t", db);
    EXPECT_EQ(result->rows[0][0], "3");
    FreeQueryResult(result);
}

/* UPDATE 更新 */
TEST_F(E2EDMLTest, UpdateRow) {
    execute_sql("INSERT INTO t VALUES (1, 'old')", db);
    execute_sql("UPDATE t SET val = 'new' WHERE id = 1", db);
    
    QueryResult *result = execute_sql("SELECT val FROM t WHERE id = 1", db);
    EXPECT_STREQ(result->rows[0][0], "new");
    FreeQueryResult(result);
}

/* DELETE 删除 */
TEST_F(E2EDMLTest, DeleteRow) {
    execute_sql("INSERT INTO t VALUES (1, 'x')", db);
    execute_sql("INSERT INTO t VALUES (2, 'y')", db);
    execute_sql("DELETE FROM t WHERE id = 1", db);
    
    QueryResult *result = execute_sql("SELECT COUNT(*) FROM t", db);
    EXPECT_EQ(result->rows[0][0], "1");
    FreeQueryResult(result);
}

} /* extern "C" */
```
